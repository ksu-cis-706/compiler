045/**
046 * This class is used to type check a StaticJava {@link CompilationUnit} with a
047 * given {@link SymbolTable}.
048 * 
049 * @author <a href="mailto:robby@ksu.edu">Robby</a>
050 */
051public class TypeChecker {
052  /**
053   * This class is used to signal an error in the process of type checking. It
054   * contains the {@link ASTNode} that causes the error.
055   * 
056   * @author <a href="mailto:robby@ksu.edu">Robby</a>
057   */
058  public static class Error extends RuntimeException {
059    private static final long serialVersionUID = 5322515711212690043L;
060    
061    /**
062     * Holds the {@link ASTNode} that causes this error.
063     */
064    public final @NonNull
065    @ReadOnly
066    ASTNode node;
067
068    /**
069     * Constructs an error from a type checker.
070     * 
071     * @param node
072     *          The {@link ASTNode} that causes this error.
073     * @param msg
074     *          The message that indicates the cause of this error.
075     */
076    public Error(@NonNull final ASTNode node, @NonNull final String msg) {
077      super(msg);
078      assert (msg != null) && (node != null);
079      this.node = node;
080    }
081  }

083  /**
084   * The visitor for {@link ASTNode} to type check a StaticJava
085   * {@link CompilationUnit}.
086   * 
087   * @author <a href="mailto:robby@ksu.edu">Robby</a>
088   */
089  protected static class Visitor extends ASTVisitor {
090    /**
091     * Holds the symbol map of the StaticJava {@link CompilationUnit}.
092     */
093    protected @NonNullElements
094    Map<ASTNode, Object> symbolMap;
095
096    /**
097     * Holds a map of {@link ASTNode} to its {@link Type}.
098     */
099    public @NonNullElements
100    Map<ASTNode, Type> resultTypeMap;
101
102    /**
103     * Holds a map of {@link MethodDeclaration} to its return {@link Type} and
104     * its parameter {@link Type}s.
105     */
106    public @NonNullElements
107    Map<Object, Pair<Type, List<Type>>> resultMethodTypeMap;
108
109    /**
110     * The {@link TypeFactory}.
111     */
112    protected @NonNull
113    TypeFactory tf;
114
115    /**
116     * Holds the string {@link ArrayType}.
117     */
118    protected @NonNull
119    ArrayType stringArrayType;
120
121    /**
122     * Holds the main class of this StaticJava {@link CompilationUnit}.
123     */
124    protected String className;
125
126    /**
127     * Holds a method's return {@link Type}.
128     */
129    protected Type methodReturnType = null;
130
131    /**
132     * Holds an expression's {@link Type}.
133     */
134    protected Type resultingType = null;
135
136    /**
137     * Constructs a visitor to type check a StaticJava {@link CompilationUnit}
138     * with the given {@link SymbolTable} and the given {@link TypeFactory}.
139     * 
140     * @param tf
141     *          The {@link TypeFactory}.
142     * @param st
143     *          The {@link SymbolTable} for the {@link CompilationUnit}.
144     */
145    protected Visitor(@NonNull final TypeFactory tf,
146        @NonNull final SymbolTable st) {
147      this.tf = tf;
148      this.symbolMap = st.symbolMap;
149      this.resultTypeMap = new HashMap<>();
150      this.resultMethodTypeMap = new HashMap<>();
151      this.stringArrayType = tf.getArrayType(tf
152          .getClassType("java.lang.String"));
153    }
154
155    protected Type convertType(final ASTNode node,
156        @SuppressWarnings("rawtypes") final Class c) {
157      if (c == void.class) {
158        return this.tf.Void;
159      }
160      if (c == boolean.class) {
161        return this.tf.Boolean;
162      } else if (c == int.class) {
163        return this.tf.Int;
164      } else if (c == String[].class) {
165        return this.stringArrayType;
166      } else if (c.isPrimitive()) {
167        throw new Error(null, "Unexpected primitive type (Class): \'"
168            + c.getName() + "\'");
169      } else if (!c.isArray()) {
170        return this.tf.getClassType(c.getName());
171      } else {
172        return this.tf.getArrayType((BaseType) convertType(
173            node,
174            c.getComponentType()));
175      }
176    }
177
178    protected Type convertType(final ASTNode node,
179        final org.eclipse.jdt.core.dom.Type t) {
180      if (t instanceof PrimitiveType) {
181        final PrimitiveType.Code ptc = ((PrimitiveType) t)
182            .getPrimitiveTypeCode();
183        if (ptc == PrimitiveType.BOOLEAN) {
184          return this.tf.Boolean;
185        } else if (ptc == PrimitiveType.INT) {
186          return this.tf.Int;
187        } else if (ptc == PrimitiveType.VOID) {
188          return this.tf.Void;
189        }
190      } else if (t instanceof SimpleType) {
191        final SimpleType st = (SimpleType) t;
192        final String name = st.getName().getFullyQualifiedName();
193        if ("String".equals(name)
194            || "java.lang.String".equals(name)) {
195          return this.stringArrayType.baseType;
196        }
197      } else if (t instanceof org.eclipse.jdt.core.dom.ArrayType) {
198        final org.eclipse.jdt.core.dom.ArrayType at = (org.eclipse.jdt.core.dom.ArrayType) t;
199        final BaseType bt = (BaseType) convertType(node, at.getElementType());
200        return this.tf.getArrayType(bt);
201      }
202      throw new Error(null, "Unexpected Type: \'" + t + "\'");
203    }
204
205    @SuppressWarnings("rawtypes")
206    protected Class convertType(final ASTNode node, final Type t) {
207      if (t == this.tf.Void) {
208        return void.class;
209      } else if (t == this.tf.Boolean) {
210        return boolean.class;
211      } else if (t == this.tf.Int) {
212        return int.class;
213      } else if (t == this.stringArrayType) {
214        return String[].class;
215      } else if (t instanceof ClassType) {
216        try {
217          return Class.forName(((ClassType) t).name);
218        } catch (final ClassNotFoundException e) {
219          throw new Error(node, "Unresolvable Type: \'" + t + "\'");
220        }
221      } else if (t instanceof ArrayType) {
222        final ArrayType at = (ArrayType) t;
223        final BaseType bt = at.baseType;
224        if (bt == this.tf.Boolean) {
225          return boolean[].class;
226        } else if (bt == this.tf.Int) {
227          return int[].class;
228        } else if (bt instanceof ClassType) {
229          try {
230            return Class.forName("[L" + ((ClassType) bt).name + ";");
231          } catch (final ClassNotFoundException e) {
232            throw new Error(node, "Unresolvable Type: \'" + t + "\'");
233          }
234        } else {
235          throw new Error(node, "Unexpected BaseType: \'" + t + "\'");
236        }
237      } else {
238        throw new Error(node, "Unexpected Type: \'" + t + "\'");
239      }
240    }
241
242    protected void dispose() {
243      this.symbolMap = null;
244      this.resultTypeMap = null;
245      this.tf = null;
246      this.stringArrayType = null;
247      this.className = null;
248      this.methodReturnType = null;
249      this.resultingType = null;
250    }
251
252    protected Type getResult() {
253      final Type result = this.resultingType;
254      this.resultingType = null;
255      return result;
256    }
257
258    @SuppressWarnings("unchecked")
259    protected Method resolveMethod(final MethodInvocation node,
260        final String className, final String methodName, final Type[] argTypes) {
261      try {
262        final int numOfArgs = argTypes.length;
263        @SuppressWarnings("rawtypes")
264        final Class[] paramTypes = new Class[numOfArgs];
265        for (int i = 0; i < numOfArgs; i++) {
266          paramTypes[i] = convertType(node, argTypes[i]);
267        }
268        @SuppressWarnings("rawtypes")
269        final Class c = Class.forName(className);
270        final Method m = c.getMethod(methodName, paramTypes);
271        this.symbolMap.put(node, m);
272        return m;
273      } catch (final ClassNotFoundException e) {
274        throw new Error(node, "Unresolvable class \"" + className + "\"");
275      } catch (final SecurityException | NoSuchMethodException e) {
276        throw new Error(node, "Unresolvable method \"" + className + "."
277            + methodName + "\"");
278      }
279    }
280
281    protected void setResult(final Expression e, final Type t) {
282      assert t != null;
283      this.resultTypeMap.put(e, t);
284      this.resultingType = t;
285    }
286
287    protected void typeCheckMethodInvocation(final MethodInvocation node,
288        final String className, final String methodName, final Type[] argTypes,
289        final Method m) {
290      @SuppressWarnings("rawtypes")
291      final Class[] paramTypeClasses = m.getParameterTypes();
292      final int numOfParams = paramTypeClasses.length;
293      if (argTypes.length != numOfParams) {
294        throw new Error(node, "Wrong number of arguments to invoke method \""
295            + methodName + "\" in \"" + node + "\"");
296      }
297      final List<Type> paramTypes = new ArrayList<>();
298      for (int i = 0; i < numOfParams; i++) {
299        final Type t = convertType(node, paramTypeClasses[i]);
300        if (t != argTypes[i]) {
301          throw new Error(node, "Type mismatch the " + i + " argument in \""
302              + node + "\"");
303        }
304        paramTypes.add(t);
305      }
306      final Type returnType = convertType(node, m.getReturnType());
307      if (!this.resultMethodTypeMap.containsKey(m)) {
308        this.resultMethodTypeMap.put(m, new Pair<>(returnType,
309            paramTypes));
310      }
311      setResult(node, returnType);
312    }
313
314    protected void typeCheckMethodInvocation(final MethodInvocation node,
315        final String className, final String methodName, final Type[] argTypes,
316        final MethodDeclaration md) {
317      final int numOfParams = md.parameters().size();
318      if (argTypes.length != numOfParams) {
319        throw new Error(node, "Wrong number of arguments to invoke method \""
320            + methodName + "\" in \"" + node + "\"");
321      }
322      for (int i = 0; i < numOfParams; i++) {
323        final Type t = convertType(node, ((SingleVariableDeclaration) md
324            .parameters().get(i)).getType());
325        if (t != argTypes[i]) {
326          throw new Error(node, "Type mismatch the " + i + " argument in \""
327              + node + "\"");
328        }
329      }
330      final Type returnType = convertType(node, md.getReturnType2());
331      setResult(node, returnType);
332    }

334    @Override
335    public boolean visit(final Assignment node) {
336      node.getLeftHandSide().accept(this);
337      final Type lhsType = getResult();
338      node.getRightHandSide().accept(this);
339      final Type rhsType = getResult();
340      if (lhsType != rhsType) {
341        throw new Error(node, "Type mismatch in \"" + node + "\": " + lhsType
342            + " = " + rhsType);
343      }
344      // no need to set the type result for assignments since
345      // assignments in StaticJava are statements,
346      // i.e., they are evaluated for their side-effects.
347      return false;
348    }
349
350    @Override
351    public boolean visit(final BooleanLiteral node) {
352      setResult(node, this.tf.Boolean);
353      return false;
354    }
355
356    @Override
357    public boolean visit(final ExpressionStatement node) {
358      final Expression e = node.getExpression();
359      e.accept(this);
360      if (e instanceof Assignment) {
361        // assignment should not have a resulting type.
362        assert getResult() == null;
363      } else if (node.getExpression() instanceof MethodInvocation) {
364        // method invocation's result can be any type (including void)
365        // so we can ignore it.
366        getResult();
367      } else {
368        throw new Error(node, "Unexpected SimpleName: \'" + node + "\'");
369      }
370      return false;
371    }
372
373    @Override
374    public boolean visit(final FieldDeclaration node) {
375      this.resultTypeMap.put(node, convertType(node, node.getType()));
376      return false;
377    }
378
379    @Override
380    public boolean visit(final IfStatement node) {
381      node.getExpression().accept(this);
382      if (getResult() != this.tf.Boolean) {
383        throw new Error(node,
384            "Expecting a boolean type expression as the condition of an if-statement: \""
385                + node.getExpression() + "\"");
386      }
387      node.getThenStatement().accept(this);
388      node.getElseStatement().accept(this);
389      return false;
390    }
391
392    @Override
393    public boolean visit(final InfixExpression node) {
394      node.getLeftOperand().accept(this);
395      final Type lhsType = getResult();
396      node.getRightOperand().accept(this);
397      final Type rhsType = getResult();
398      final InfixExpression.Operator op = node.getOperator();
399      if ((op == InfixExpression.Operator.TIMES)
400          || (op == InfixExpression.Operator.DIVIDE)
401          || (op == InfixExpression.Operator.REMAINDER)
402          || (op == InfixExpression.Operator.PLUS)
403          || (op == InfixExpression.Operator.MINUS)) {
404        if (lhsType != this.tf.Int) {
405          throw new Error(node,
406              "Expecting an int type expression as the left-hand operand of \""
407                  + op + "\" in \"" + node + "\"");
408        }
409        if (rhsType != this.tf.Int) {
410          throw new Error(node,
411              "Expecting an int type expression as the right-hand operand of \""
412                  + op + "\" in \"" + node + "\"");
413        }
414        setResult(node, this.tf.Int);
415      } else if ((op == InfixExpression.Operator.LESS)
416          || (op == InfixExpression.Operator.GREATER)
417          || (op == InfixExpression.Operator.LESS_EQUALS)
418          || (op == InfixExpression.Operator.GREATER_EQUALS)) {
419        if (lhsType != this.tf.Int) {
420          throw new Error(node,
421              "Expecting an int type expression as the left-hand operand of \""
422                  + op + "\" in \"" + node + "\"");
423        }
424        if (rhsType != this.tf.Int) {
425          throw new Error(node,
426              "Expecting an int type expression as the right-hand operand of \""
427                  + op + "\" in \"" + node + "\"");
428        }
429        setResult(node, this.tf.Boolean);
430      } else if ((op == InfixExpression.Operator.CONDITIONAL_AND)
431          || (op == InfixExpression.Operator.CONDITIONAL_OR)) {
432        if (lhsType != this.tf.Boolean) {
433          throw new Error(node,
434              "Expecting a boolean type expression as the left-hand operand of \""
435                  + op + "\" in \"" + node + "\"");
436        }
437        if (rhsType != this.tf.Boolean) {
438          throw new Error(node,
439              "Expecting a boolean type expression as the right-hand operand of \""
440                  + op + "\" in \"" + node + "\"");
441        }
442        setResult(node, this.tf.Boolean);
443      } else if ((op == InfixExpression.Operator.EQUALS)
444          || (op == InfixExpression.Operator.NOT_EQUALS)) {
445        if (lhsType != rhsType) {
446          throw new Error(node, "Type mismatch in \"" + node + "\": " + lhsType
447              + " " + op + " " + rhsType);
448        }
449        setResult(node, this.tf.Boolean);
450      } else {
451        throw new Error(node, "Unexpected InfixExpression: \'" + node + "\'");
452      }
453      return false;
454    }

456    @Override
457    public boolean visit(final MethodDeclaration node) {
458      this.methodReturnType = convertType(node, node.getReturnType2());
459      final List<Type> paramTypes = new ArrayList<>();
460      for (final Object o : node.parameters()) {
461        final SingleVariableDeclaration sdv = (SingleVariableDeclaration) o;
462        final Type t = convertType(node, sdv.getType());
463        paramTypes.add(t);
464        this.resultTypeMap.put(sdv, t);
465      }
466      this.resultMethodTypeMap.put(node, new Pair<>(
467          this.methodReturnType, paramTypes));
468      node.getBody().accept(this);
469      this.methodReturnType = null;
470      return false;
471    }
472
473    @Override
474    public boolean visit(final MethodInvocation node) {
475      // note that we don't visit the MethodInvocation's simple name
476      // because we want visit(SimpleName) to resolve variable references
477      // instead of method names
478      final String className = node.getExpression() == null ? this.className
479          : ((SimpleName) node.getExpression()).getIdentifier();
480      final String methodName = node.getName().getIdentifier();
481      final int numOfArgs = node.arguments().size();
482      final Type[] argTypes = new Type[numOfArgs];
483      for (int i = 0; i < numOfArgs; i++) {
484        ((Expression) node.arguments().get(i)).accept(this);
485        argTypes[i] = getResult();
486      }
487      final Object o = this.symbolMap.get(node);
488      if ((o == null) || (o instanceof Method)) {
489        final Method m = o == null ? resolveMethod(
490            node,
491            className,
492            methodName,
493            argTypes) : (Method) o;
494        typeCheckMethodInvocation(node, className, methodName, argTypes, m);
495      } else {
496        typeCheckMethodInvocation(
497            node,
498            className,
499            methodName,
500            argTypes,
501            (MethodDeclaration) o);
502      }
503      return false;
504    }

512    @Override

514      node.getExpression().accept(this);
515      setResult(node, getResult());
516      return false;

519    @Override
520    public boolean visit(final PrefixExpression node) {
521      node.getOperand().accept(this);
522      final Type t = getResult();
523      final PrefixExpression.Operator op = node.getOperator();
524      if ((op == PrefixExpression.Operator.PLUS)
525          || (op == PrefixExpression.Operator.MINUS)) {
526        if (t != this.tf.Int) {
527          throw new Error(node,
528              "Expecting an int type expression as the operand of \"" + op
529                  + "\" in \"" + node + "\"");
530        }
531        setResult(node, this.tf.Int);
532      } else if (op == PrefixExpression.Operator.NOT) {
533        if (t != this.tf.Boolean) {
534          throw new Error(node,
535              "Expecting a boolean type expression as the operand of \"" + op
536                  + "\" in \"" + node + "\"");
537        }
538        setResult(node, this.tf.Boolean);
539      } else {
540        throw new Error(node, "Unexpected PrefixExpression: \'" + node + "\'");
541      }
542      return false;
543    }
544
545    @Override
546    public boolean visit(final ReturnStatement node) {
547      final Expression e = node.getExpression();
548      if ((this.methodReturnType == this.tf.Void) && (e != null)) {
549        throw new Error(node, "Unexpected return's expression in \"" + node
550            + "\"");
551      } else if ((this.methodReturnType != this.tf.Void) && (e == null)) {
552        throw new Error(node, "Expecting a return's expression in \"" + node
553            + "\"");
554      } else if ((this.methodReturnType != this.tf.Void) && (e != null)) {
555        e.accept(this);
556        final Type t = getResult();
557        if (t != this.methodReturnType) {
558          throw new Error(node, "Expecting " + this.methodReturnType.name
559              + " return expression in \"" + node + "\"");
560        }
561      }
562      return false;
563    }
564
565    @Override
566    public boolean visit(final SimpleName node) {
567      final ASTNode parent = node.getParent();
568      if ((parent instanceof Expression) || (parent instanceof Statement)) {
569        final Object o = this.symbolMap.get(node);
570        if (o instanceof FieldDeclaration) {
571          final FieldDeclaration fd = (FieldDeclaration) o;
572          setResult(node, convertType(node, fd.getType()));
573        } else if (o instanceof SingleVariableDeclaration) {
574          final SingleVariableDeclaration svd = (SingleVariableDeclaration) o;
575          setResult(node, convertType(node, svd.getType()));
576        } else if (o instanceof VariableDeclarationStatement) {
577          final VariableDeclarationStatement vds = (VariableDeclarationStatement) o;
578          setResult(node, convertType(node, vds.getType()));
579        } else {
580          throw new Error(node, "Unexpected SimpleName: \'" + node + "\'");
581        }
582      }
583      return false;
584    }
585
586    @Override
587    public boolean visit(final TypeDeclaration node) {
588      this.className = node.getName().getIdentifier();
589      this.resultTypeMap.put(node, this.tf.getClassType(this.className));
590      return super.visit(node);
591    }
592
593    @Override
594    public boolean visit(final VariableDeclarationStatement node) {
595      this.resultTypeMap.put(node, convertType(node, node.getType()));
596      return false;
597    }
598
599    @Override
600    public boolean visit(final WhileStatement node) {
601      node.getExpression().accept(this);
602      final Type t = getResult();
603      if (t != this.tf.Boolean) {
604        throw new Error(node,
605            "Expecting a boolean type expression as the condition of a while-statement: \""
606                + node.getExpression() + "\"");
607      }
608      node.getBody().accept(this);
609      return false;
610    }

613  /**
614   * Type checks a StaticJava {@link CompilationUnit} with the given
615   * {@link SymbolTable} and the given {@link TypeFactory}. It also resolves
616   * {@link MethodInvocation} of library call (and put its mapping in the
617   * {@link SymbolTable}).
618   * 
619   * @param tf
620   *          The {@link TypeFactory}.
621   * @param cu
622   *          The StaticJava {@link CompilationUnit}.
623   * @param symbolTable
624   *          The {@link SymbolTable} of the {@link CompilationUnit}
625   * @return The {@link TypeTable}.
626   * @throws TypeChecker.Error
627   *           If the type checker encounter type error in the
628   *           {@link CompilationUnit}.
629   */
630  public static @NonNull
631  TypeTable check(@NonNull final TypeFactory tf,
632      @NonNull final CompilationUnit cu, @NonNull final SymbolTable symbolTable)
633      throws TypeChecker.Error {
634    assert (tf != null) && (cu != null) && (symbolTable != null);
635
636    final Visitor v = new Visitor(tf, symbolTable);
637    cu.accept(v);
638    final TypeTable result = new TypeTable(v.resultTypeMap,
639        v.resultMethodTypeMap);
640    v.dispose();
641    return result;
642  }
