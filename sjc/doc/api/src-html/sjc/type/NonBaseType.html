001package sjc.type;
002
003import sjc.annotation.NonNull;
004
005/**
006 * This class represents a non-base type of an {@link ArrayType} in StaticJava.
007 * 
008 * @author <a href="mailto:robby@ksu.edu">Robby</a>
009 */
010public abstract class NonBaseType extends Type {
011  /**
012   * Constucts a base type with the given name.
013   * 
014   * @param name
015   *          The name of this base type.
016   */
017  protected NonBaseType(@NonNull final String name) {
018    super(name);
019  }
020}
