001package sjc.type;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import sjc.annotation.NonNullElements;
007import sjc.annotation.ReadOnly;
008
009/**
010 * This class represents a class type in StaticJava.
011 * 
012 * @author <a href="mailto:robby@ksu.edu">Robby</a>
013 */
014public class ClassType extends BaseType
015    implements NonPrimitiveType {
016  /**
017   * Holds a map of field names to their {@link Type}.
018   */
019  public final @NonNullElements
020  @ReadOnly
021  Map<String, Type> fieldTypeMap = new HashMap<>();
022
023  /**
024   * Constructs a class type with the given fully-qualified name.
025   * 
026   * @param name
027   *          The fully qualified name of this class type.
028   */
029  protected ClassType(final String name) {
030    super(name);
031  }
032}
