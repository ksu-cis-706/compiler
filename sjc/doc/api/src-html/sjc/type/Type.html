001package sjc.type;
002
003import sjc.annotation.NonNull;
004import sjc.annotation.PossiblyNull;
005import sjc.annotation.ReadOnly;
006
007/**
008 * This class represents a type in StaticJava.
009 * 
010 * @author <a href="mailto:robby@ksu.edu">Robby</a>
011 */
012public abstract class Type {
013  /**
014   * Holds this type's name.
015   */
016  public final @NonNull
017  @ReadOnly
018  String name;
019
020  /**
021   * Constructs a type with the given name.
022   * 
023   * @param name
024   *          The name of this type.
025   */
026  protected Type(@NonNull final String name) {
027    assert name != null;
028    this.name = name;
029  }
030
031  /**
032   * Determines whether a given object is equal to this type.
033   * 
034   * @param o
035   *          The other object for comparison.
036   * @return True, if the o object is a type with the same name as this type's.
037   */
038  @Override
039  public boolean equals(@PossiblyNull final Object o) {
040    return o instanceof Type && toString().equals(o.toString());
041  }
042
043  /**
044   * Returns the hash code of this type. Type with the same names have the same
045   * hash code.
046   * 
047   * @return The hash code of this type.
048   */
049  @Override
050  public int hashCode() {
051    return toString().hashCode();
052  }
053
054  /**
055   * Returns the {@link String} representation of this type, i.e., its name.
056   * 
057   * @return The {@link String} representation of this type, i.e., its name.
058   */
059  @Override
060  public @NonNull
061  String toString() {
062    return this.name;
063  }
064}
