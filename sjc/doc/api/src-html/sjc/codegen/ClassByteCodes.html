001package sjc.codegen;
002
003import sjc.annotation.NonNull;
004
005/**
006 * This class is used to hold the result of the {@link ByteCodeGenerator} (when
007 * the generation is succesful).
008 * 
009 * @author <a href="mailto:robby@ksu.edu">Robby</a>
010 */
011public class ClassByteCodes {
012  /**
013   * Holds the name of the main class.
014   */
015  public final @NonNull
016  String mainClassName;
017
018  /**
019   * Holds the byte array representation of the main class.
020   */
021  public final @NonNull
022  byte[] mainClassBytes;
023
024  /**
025   * Constructs the class byte codes.
026   * 
027   * @param mainClassName
028   *          The name of the main class.
029   * @param mainClassBytes
030   *          The byte array representation of the main class.
031   */
032  public ClassByteCodes(@NonNull final String mainClassName,
033      @NonNull final byte[] mainClassBytes) {
034    assert (mainClassName != null) && (mainClassBytes != null);
035    this.mainClassName = mainClassName;
036    this.mainClassBytes = mainClassBytes;
037  }
038}
