056/**
057 * This class is used to translate a StaticJava {@link CompilationUnit} to
058 * {@link ClassByteCodes} that represent a Java 1.5 class files.
059 *
060 * @author <a href="mailto:robby@ksu.edu">Robby</a>
061 */
062public class ByteCodeGenerator {
063  /**
064   * This class is used to signal an error in the process of generating
065   * bytecode. It contains the {@link ASTNode} that causes the error.
066   *
067   * @author <a href="mailto:robby@ksu.edu">Robby</a>
068   */
069  public static class Error extends RuntimeException {
070    private static final long serialVersionUID = 2821914074630252345L;
071
072    /**
073     * Holds the {@link ASTNode} that causes this error.
074     */
075    public final @NonNull @ReadOnly ASTNode node;
076
077    /**
078     * Constructs an error of bytecode generation.
079     *
080     * @param node
081     *          The {@link ASTNode} that causes this error.
082     * @param msg
083     *          The message that indicates the cause of this error.
084     */
085    public Error(@NonNull final ASTNode node, @NonNull final String msg) {
086      super(msg);
087      assert (msg != null) && (node != null);
088      this.node = node;
089    }

092  /**
093   * The visitor for {@link ASTNode} to generate bytecodes.
094   *
095   * @author <a href="mailto:robby@ksu.edu">Robby</a>
096   */
097  protected static class Visitor extends ASTVisitor {
098    protected @NonNull ClassWriter cw;
099
100    protected @PossiblyNull FieldVisitor fv;
101
102    protected @PossiblyNull MethodVisitor mv;
103
104    public @PossiblyNull String mainClassName;
105
106    public @PossiblyNull byte[] mainClassBytes;
107
108    protected @NonNullElements @ReadOnlyElements Map<ASTNode, Object> symbolMap;
109
110    protected @NonNullElements @ReadOnlyElements Map<ASTNode, Type> typeMap;
111
112    protected @NonNullElements @ReadOnlyElements Map<Object, Pair<Type, List<Type>>> methodTypeMap;
113
114    protected @NonNull @NonNullElementsOnly List<Pair<String, Type>> localNamesTypes = new ArrayList<>();
115
116    protected @NonNull @NonNullElementsOnly Map<String, Integer> localIndexMap = new HashMap<>();
117
118    protected Visitor(@NonNull final SymbolTable st, @NonNull final TypeTable tt) {
119      assert (st != null) && (tt != null);
120      this.symbolMap = st.symbolMap;
121      this.typeMap = tt.typeMap;
122      this.methodTypeMap = tt.methodTypeMap;
123    }
124
125    protected void buildLocalIndexTable(final MethodDeclaration md) {
126      int i = 0;
127      for (final Object o : md.parameters()) {
128        final SingleVariableDeclaration svd = (SingleVariableDeclaration) o;
129        final String paramName = svd.getName().getIdentifier();
130        this.localNamesTypes.add(new Pair<>(paramName, this.typeMap
131            .get(svd)));
132        this.localIndexMap.put(paramName, i);
133        i++;
134      }
135      for (final Object o : md.getBody().statements()) {
136        if (!(o instanceof VariableDeclarationStatement)) {
137          break;
138        }
139        final VariableDeclarationStatement vds = (VariableDeclarationStatement) o;
140        final VariableDeclarationFragment vdf = (VariableDeclarationFragment) vds
141            .fragments().get(0);
142        final String localName = vdf.getName().getIdentifier();
143        this.localNamesTypes.add(new Pair<>(localName, this.typeMap
144            .get(vds)));
145        this.localIndexMap.put(localName, i);
146        i++;
147      }
148    }
149
150    @SuppressWarnings("rawtypes")
151    protected int convertModifiers(final ASTNode node, final List l) {
152      int result = 0;
153      for (final Object o : l) {
154        if (o instanceof Modifier) {
155          final ModifierKeyword mk = ((Modifier) o).getKeyword();
156          if (mk == ModifierKeyword.PUBLIC_KEYWORD) {
157            result += Opcodes.ACC_PUBLIC;
158          } else if (mk == ModifierKeyword.STATIC_KEYWORD) {
159            result += Opcodes.ACC_STATIC;
160          } else {
161            throw new Error(node, "Unexpected modifier \'" + o + "\' in \'"
162                + node.getParent() + "\'");
163          }
164        }
165      }
166      return result;
167    }
168
169    protected String convertType(final Type t) {
170      if (t instanceof VoidType) {
171        return "V";
172      } else if (t instanceof BooleanType) {
173        return "Z";
174      } else if (t instanceof IntType) {
175        return "I";
176      } else if (t instanceof ClassType) {
177        return "L" + ((ClassType) t).name.replace('.', '/') + ";";
178      } else {
179        return "[" + convertType(((ArrayType) t).baseType);
180      }
181    }
182
183    protected void dispose() {
184      this.cw = null;
185      this.symbolMap = null;
186      this.typeMap = null;
187      this.methodTypeMap = null;
188      this.localNamesTypes = null;
189      this.localIndexMap = null;
190      this.mainClassName = null;
191      this.mainClassBytes = null;
192    }
193
194    protected void generateConstructor(final String className) {
195      this.mv = this.cw.visitMethod(
196          Opcodes.ACC_PUBLIC,
197          "<init>",
198          "()V",
199          null,
200          null);
201      this.mv.visitCode();
202      final Label l0 = new Label();
203      this.mv.visitLabel(l0);
204      this.mv.visitVarInsn(Opcodes.ALOAD, 0);
205      this.mv.visitMethodInsn(
206          Opcodes.INVOKESPECIAL,
207          "java/lang/Object",
208          "<init>",
209          "()V",
210          false);
211      this.mv.visitInsn(Opcodes.RETURN);
212      final Label l1 = new Label();
213      this.mv.visitLabel(l1);
214      this.mv
215      .visitLocalVariable("this", "L" + className + ";", null, l0, l1, 0);
216      this.mv.visitMaxs(1, 1);
217      this.mv.visitEnd();
218    }
219
220    protected void generateIntConst(final int i) {
221      switch (i) {
222        case -1:
223          this.mv.visitInsn(Opcodes.ICONST_M1);
224          break;
225        case 0:
226          this.mv.visitInsn(Opcodes.ICONST_0);
227          break;
228        case 1:
229          this.mv.visitInsn(Opcodes.ICONST_1);
230          break;
231        case 2:
232          this.mv.visitInsn(Opcodes.ICONST_2);
233          break;
234        case 3:
235          this.mv.visitInsn(Opcodes.ICONST_3);
236          break;
237        case 4:
238          this.mv.visitInsn(Opcodes.ICONST_4);
239          break;
240        case 5:
241          this.mv.visitInsn(Opcodes.ICONST_5);
242          break;
243        default:
244          if ((i >= Byte.MIN_VALUE) && (i <= Byte.MAX_VALUE)) {
245            this.mv.visitIntInsn(Opcodes.BIPUSH, i);
246          } else if ((i >= Short.MIN_VALUE) && (i <= Short.MAX_VALUE)) {
247            this.mv.visitIntInsn(Opcodes.SIPUSH, i);
248          } else {
249            this.mv.visitLdcInsn(i);
250          }
251      }
252    }
253
254    protected void generateRelationalCode(final int opcode) {
255      final Label thenLabel = new Label();
256      final Label endLabel = new Label();
257      this.mv.visitJumpInsn(opcode, thenLabel);
258      this.mv.visitInsn(Opcodes.ICONST_0);
259      this.mv.visitJumpInsn(Opcodes.GOTO, endLabel);
260      this.mv.visitLabel(thenLabel);
261      this.mv.visitInsn(Opcodes.ICONST_1);
262      this.mv.visitLabel(endLabel);
263    }
264
265    protected String getMethodDescriptor(final Type returnType,
266        final List<Type> parameterTypes) {
267      final StringBuilder sb = new StringBuilder();
268      sb.append('(');
269      for (final Type t : parameterTypes) {
270        sb.append(convertType(t));
271      }
272      sb.append(')');
273      sb.append(convertType(returnType));
274      return sb.toString();
275    }

277    @Override
278    public boolean visit(final Assignment node) {
279      node.getRightHandSide().accept(this);
280      final ASTNode lhsNode = node.getLeftHandSide();
281      final String varName = ((SimpleName) lhsNode).getIdentifier();
282      final Object lhsDecl = this.symbolMap.get(lhsNode);
283      if (lhsDecl instanceof FieldDeclaration) {
284        final FieldDeclaration fd = (FieldDeclaration) lhsDecl;
285        final String className = ((TypeDeclaration) fd.getParent()).getName()
286            .getIdentifier();
287        this.mv.visitFieldInsn(
288            Opcodes.PUTSTATIC,
289            className,
290            varName,
291            convertType(this.typeMap.get(fd)));
292      } else {
293        this.mv.visitVarInsn(Opcodes.ISTORE, this.localIndexMap.get(varName));
294      }
295      return false;
296    }

298    @Override
299    public boolean visit(final BooleanLiteral node) {
300      if (node.booleanValue()) {
301        this.mv.visitInsn(Opcodes.ICONST_1);
302      } else {
303        this.mv.visitInsn(Opcodes.ICONST_0);
304      }
305      return false;
306    }

308    @Override
309    public boolean visit(final ExpressionStatement node) {
310      final Expression e = node.getExpression();
311      e.accept(this);
312      final Type t = this.typeMap.get(e);
313      if ((t == null) || (t instanceof VoidType)) {
314        // skip
315      } else {
316        this.mv.visitInsn(Opcodes.POP);
317      }
318      return false;
319    }

321    @Override
322    public boolean visit(final FieldDeclaration node) {
323      final int modifiers = convertModifiers(node, node.modifiers());
324      final VariableDeclarationFragment vdf = (VariableDeclarationFragment) node
325          .fragments().get(0);
326      this.cw.visitField(
327          modifiers,
328          vdf.getName().getIdentifier(),
329          convertType(this.typeMap.get(node)),
330          null,
331          null).visitEnd();
332
333      return false;
334    }
335
336    @Override
337    public boolean visit(final IfStatement node) {
338      final Label elseOrEndLabel = new Label();
339      node.getExpression().accept(this);
340      this.mv.visitJumpInsn(Opcodes.IFEQ, elseOrEndLabel);
341      node.getThenStatement().accept(this);
342      final Block elseBlock = (Block) node.getElseStatement();
343      if (elseBlock == null) {
344        this.mv.visitLabel(elseOrEndLabel);
345      } else {
346        final Label endLabel = new Label();
347        this.mv.visitJumpInsn(Opcodes.GOTO, endLabel);
348        this.mv.visitLabel(elseOrEndLabel);
349        elseBlock.accept(this);
350        this.mv.visitLabel(endLabel);
351      }
352      return false;
353    }

355    @Override
356    public boolean visit(final InfixExpression node) {
357      final InfixExpression.Operator op = node.getOperator();
358      node.getLeftOperand().accept(this);
359      if (op == InfixExpression.Operator.CONDITIONAL_AND) {
360        this.mv.visitInsn(Opcodes.DUP);
361        final Label falseLabel = new Label();
362        this.mv.visitJumpInsn(Opcodes.IFEQ, falseLabel);
363        this.mv.visitInsn(Opcodes.POP);
364        node.getRightOperand().accept(this);
365        this.mv.visitLabel(falseLabel);
366      } else if (op == InfixExpression.Operator.CONDITIONAL_OR) {
367        this.mv.visitInsn(Opcodes.DUP);
368        final Label trueLabel = new Label();
369        this.mv.visitJumpInsn(Opcodes.IFNE, trueLabel);
370        this.mv.visitInsn(Opcodes.POP);
371        node.getRightOperand().accept(this);
372        this.mv.visitLabel(trueLabel);
373      } else {
374        node.getRightOperand().accept(this);
375        if (op == InfixExpression.Operator.PLUS) {
376          this.mv.visitInsn(Opcodes.IADD);
377        } else if (op == InfixExpression.Operator.MINUS) {
378          this.mv.visitInsn(Opcodes.ISUB);
379        } else if (op == InfixExpression.Operator.TIMES) {
380          this.mv.visitInsn(Opcodes.IMUL);
381        } else if (op == InfixExpression.Operator.DIVIDE) {
382          this.mv.visitInsn(Opcodes.IDIV);
383        } else if (op == InfixExpression.Operator.REMAINDER) {
384          this.mv.visitInsn(Opcodes.IREM);
385        } else if (op == InfixExpression.Operator.GREATER) {
386          generateRelationalCode(Opcodes.IF_ICMPGT);
387        } else if (op == InfixExpression.Operator.GREATER_EQUALS) {
388          generateRelationalCode(Opcodes.IF_ICMPGE);
389        } else if (op == InfixExpression.Operator.EQUALS) {
390          generateRelationalCode(Opcodes.IF_ICMPEQ);
391        } else if (op == InfixExpression.Operator.NOT_EQUALS) {
392          generateRelationalCode(Opcodes.IF_ICMPNE);
393        } else if (op == InfixExpression.Operator.LESS) {
394          generateRelationalCode(Opcodes.IF_ICMPLT);
395        } else if (op == InfixExpression.Operator.LESS_EQUALS) {
396          generateRelationalCode(Opcodes.IF_ICMPLE);
397        }
398      }
399      return false;
400    }

402    @Override
403    public boolean visit(final MethodDeclaration node) {
404      final String methodName = node.getName().getIdentifier();
405      final int modifiers = convertModifiers(node, node.modifiers());
406      final Pair<Type, List<Type>> methodType = this.methodTypeMap.get(node);
407      final String methodDesc = getMethodDescriptor(
408          methodType.first,
409          methodType.second);
410      this.mv = this.cw.visitMethod(
411          modifiers,
412          methodName,
413          methodDesc,
414          null,
415          null);
416      this.mv.visitCode();
417      final Label initLabel = new Label();
418      this.mv.visitLabel(initLabel);
419
420      buildLocalIndexTable(node);
421
422      Statement lastStatement = null;
423      for (final Object o : node.getBody().statements()) {
424        if (!(o instanceof VariableDeclarationStatement)) {
425          ((ASTNode) o).accept(this);
426        }
427        lastStatement = (Statement) o;
428      }
429
430      // TODO: this is a hack, we need an analysis to see whether we need
431      // a return statement or not.
432      if (!(lastStatement instanceof ReturnStatement)) {
433        if (methodType.first instanceof VoidType) {
434          this.mv.visitInsn(Opcodes.RETURN);
435        }
436      }
437
438      final Label endLabel = new Label();
439      this.mv.visitLabel(endLabel);
440
441      int i = 0;
442      for (final Pair<String, Type> p : this.localNamesTypes) {
443        this.mv.visitLocalVariable(
444            p.first,
445            convertType(p.second),
446            null,
447            initLabel,
448            endLabel,
449            i);
450        i++;
451      }
452
453      this.mv.visitMaxs(0, 0);
454      this.mv.visitEnd();
455
456      this.localNamesTypes.clear();
457      this.localIndexMap.clear();
458      return false;
459    }

461    @Override
462    public boolean visit(final MethodInvocation node) {
463      for (final Object o : node.arguments()) {
464        ((ASTNode) o).accept(this);
465      }
466      final Object o = this.symbolMap.get(node);
467      if (o instanceof MethodDeclaration) {
468        final MethodDeclaration md = (MethodDeclaration) o;
469        final String className = ((TypeDeclaration) md.getParent()).getName()
470            .getIdentifier();
471        final String methodName = node.getName().getIdentifier();
472        final Pair<Type, List<Type>> p = this.methodTypeMap.get(md);
473        this.mv.visitMethodInsn(
474            Opcodes.INVOKESTATIC,
475            className,
476            methodName,
477            getMethodDescriptor(p.first, p.second),
478            false);
479      } else if (o instanceof Method) {
480        final Method m = (Method) o;
481        final String className = m.getDeclaringClass().getName()
482            .replace('.', '/');
483        final String methodName = m.getName();
484        final Pair<Type, List<Type>> p = this.methodTypeMap.get(m);
485        this.mv.visitMethodInsn(
486            Opcodes.INVOKESTATIC,
487            className,
488            methodName,
489            getMethodDescriptor(p.first, p.second),
490            false);
491      }
492      return false;
493    }

495    @Override
496    public boolean visit(final NumberLiteral node) {
497      generateIntConst(Integer.parseInt(node.getToken()));
498      return false;

501    @Override
502    public boolean visit(final PrefixExpression node) {
503      node.getOperand().accept(this);
504      final PrefixExpression.Operator op = node.getOperator();
505      if (op == PrefixExpression.Operator.PLUS) {
506        // do nothing
507      } else if (op == PrefixExpression.Operator.MINUS) {
508        this.mv.visitInsn(Opcodes.INEG);
509      } else if (op == PrefixExpression.Operator.NOT) {
510        final Label falseLabel = new Label();
511        final Label endLabel = new Label();
512        this.mv.visitJumpInsn(Opcodes.IFEQ, falseLabel);
513        this.mv.visitInsn(Opcodes.ICONST_0);
514        this.mv.visitJumpInsn(Opcodes.GOTO, endLabel);
515        this.mv.visitLabel(falseLabel);
516        this.mv.visitInsn(Opcodes.ICONST_1);
517        this.mv.visitLabel(endLabel);
518      }
519      return false;
520    }

522    @Override
523    public boolean visit(final ReturnStatement node) {
524      final Expression e = node.getExpression();
525      if (e == null) {
526        this.mv.visitInsn(Opcodes.RETURN);
527      } else {
528        e.accept(this);
529        this.mv.visitInsn(Opcodes.IRETURN);
530      }
531      return false;
532    }

534    @Override
535    public boolean visit(final SimpleName node) {
536      final ASTNode parent = node.getParent();
537      if ((parent instanceof Expression) || (parent instanceof Statement)) {
538        final String varName = node.getIdentifier();
539        final Object decl = this.symbolMap.get(node);
540        if (decl instanceof FieldDeclaration) {
541          final FieldDeclaration fd = (FieldDeclaration) decl;
542          final String className = ((TypeDeclaration) fd.getParent()).getName()
543              .getIdentifier();
544          this.mv.visitFieldInsn(
545              Opcodes.GETSTATIC,
546              className,
547              varName,
548              convertType(this.typeMap.get(fd)));
549        } else {
550          final Type t = this.typeMap.get(node);
551          if ((t instanceof IntType) || (t instanceof BooleanType)) {
552            this.mv.visitVarInsn(Opcodes.ILOAD, this.localIndexMap.get(varName));
553          } else {
554            this.mv.visitVarInsn(Opcodes.ALOAD, this.localIndexMap.get(varName));
555          }
556        }
557      }
558      return false;
559    }

561    @Override
562    public boolean visit(final TypeDeclaration node) {
563      this.mainClassName = node.getName().getIdentifier();
564      this.cw = new ClassWriter(ClassWriter.COMPUTE_FRAMES
565          | ClassWriter.COMPUTE_MAXS);
566      this.cw.visit(
567          Opcodes.V1_5,
568          Opcodes.ACC_PUBLIC + Opcodes.ACC_SUPER,
569          this.mainClassName,
570          null,
571          "java/lang/Object",
572          null);
573      this.cw.visitSource(null, null);
574      generateConstructor(this.mainClassName);
575      for (final Object o : node.bodyDeclarations()) {
576        ((ASTNode) o).accept(this);
577      }
578      this.cw.visitEnd();
579      this.mainClassBytes = this.cw.toByteArray();
580      this.cw = null;
581      return false;
582    }
583
584    @Override
585    public boolean visit(final WhileStatement node) {
586      final Label loopLabel = new Label();
587      final Label endLabel = new Label();
588      this.mv.visitLabel(loopLabel);
589      node.getExpression().accept(this);
590      this.mv.visitJumpInsn(Opcodes.IFEQ, endLabel);
591      node.getBody().accept(this);
592      this.mv.visitJumpInsn(Opcodes.GOTO, loopLabel);
593      this.mv.visitLabel(endLabel);
594      return false;
595    }

598  /**
599   * Generates a {@link ClassByteCodes} that represents the class files for the
600   * given StaticJava {@link CompilationUnit} with the given {@link SymbolTable}
601   * and {@link TypeTable}.
602   *
603   * @param cu
604   *          The StaticJava {@link CompilationUnit}.
605   * @param st
606   *          The {@link SymbolTable} of the {@link CompilationUnit}.
607   * @param tt
608   *          The {@link TypeTable} of the {@link CompilationUnit}.
609   * @return The {@link ClassByteCodes} that represents the class files for the
610   *         given StaticJava {@link CompilationUnit} with the given
611   *         {@link SymbolTable} and {@link TypeTable}.
612   * @throws ByteCodeGenerator.Error
613   *           If the generator encounter unexpected error.
614   */
615  public static @NonNull ClassByteCodes generate(
616      @NonNull final CompilationUnit cu, @NonNull final SymbolTable st,
617      @NonNull final TypeTable tt) throws ByteCodeGenerator.Error {
618    assert (cu != null) && (st != null) && (tt != null);
619
620    final Visitor v = new Visitor(st, tt);
621    cu.accept(v);
622    final ClassByteCodes result = new ClassByteCodes(v.mainClassName,
623        v.mainClassBytes);
624    v.dispose();
625    return result;
626  }
