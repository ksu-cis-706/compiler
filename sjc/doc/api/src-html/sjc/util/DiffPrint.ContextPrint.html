068  /**
069   * A Base class for printing edit scripts produced by Diff. This class divides
070   * the change list into "hunks", and calls <code>print_hunk</code> for each
071   * hunk. Various utility methods are provided as well.
072   */
073  public static abstract class Base {
074    public static char change_letter(final int inserts, final int deletes) {
075      if (inserts == 0) {
076        return 'd';
077      } else if (deletes == 0) {
078        return 'a';
079      } else {
080        return 'c';
081      }
082    }
083
084    protected PrintWriter outfile;
085    /**
086     * Set to ignore certain kinds of lines when printing an edit script. For
087     * example, ignoring blank lines or comments.
088     */
089    protected UnaryPredicate ignore = null;
090    /**
091     * Set to the lines of the files being compared.
092     */
093    protected Object[] file0, file1;
094    protected int first0, last0, first1, last1, deletes, inserts;
095
096    protected Base(final Object[] a, final Object[] b) {
097      this.file0 = a;
098      this.file1 = b;
099    }
100
101    /**
102     * Look at a hunk of edit script and report the range of lines in each file
103     * that it applies to. HUNK is the start of the hunk, which is a chain of
104     * `struct change'. The first and last line numbers of file 0 are stored in
105     * *FIRST0 and *LAST0, and likewise for file 1 in *FIRST1 and *LAST1. Note
106     * that these are internal line numbers that count from 0.
107     * 
108     * If no lines from file 0 are deleted, then FIRST0 is LAST0+1.
109     * 
110     * Also set *DELETES nonzero if any lines of file 0 are deleted and set
111     * *INSERTS nonzero if any lines of file 1 are inserted. If only ignorable
112     * lines are inserted or deleted, both are set to 0.
113     */
114
115    protected void analyze_hunk(final Diff.change hunk) {
116      int f0, l0 = 0, f1, l1 = 0, show_from = 0, show_to = 0;
117      int i;
118      Diff.change next;
119      boolean nontrivial = (this.ignore == null);
120
121      show_from = show_to = 0;
122
123      f0 = hunk.line0;
124      f1 = hunk.line1;
125
126      for (next = hunk; next != null; next = next.link) {
127        l0 = next.line0 + next.deleted - 1;
128        l1 = next.line1 + next.inserted - 1;
129        show_from += next.deleted;
130        show_to += next.inserted;
131        for (i = next.line0; (i <= l0) && !nontrivial; i++) {
132          if (!this.ignore.execute(this.file0[i])) {
133            nontrivial = true;
134          }
135        }
136        for (i = next.line1; (i <= l1) && !nontrivial; i++) {
137          if (!this.ignore.execute(this.file1[i])) {
138            nontrivial = true;
139          }
140        }
141      }
142
143      this.first0 = f0;
144      this.last0 = l0;
145      this.first1 = f1;
146      this.last1 = l1;
147
148      /* If all inserted or deleted lines are ignorable,
149      tell the caller to ignore this hunk.  */
150
151      if (!nontrivial) {
152        show_from = show_to = 0;
153      }
154
155      this.deletes = show_from;
156      this.inserts = show_to;
157    }
158
159    /**
160     * Called with the tail of the script and returns the last link that belongs
161     * together with the start of the tail.
162     */
163
164    protected Diff.change hunkfun(final Diff.change hunk) {
165      return hunk;
166    }
167
168    protected void print_1_line(final String pre, final Object linbuf) {
169      this.outfile.println(pre + linbuf.toString());
170    }
171
172    /** Print the script header which identifies the files compared. */
173    public void print_header(final String filea, final String fileb) {
174      setupOutput();
175    }
176
177    protected abstract void print_hunk(Diff.change hunk);
178
179    /**
180     * Print a pair of line numbers with SEPCHAR, translated for file FILE. If
181     * the two numbers are identical, print just one number.
182     * 
183     * Args A and B are internal line numbers. We print the translated (real)
184     * line numbers.
185     */
186
187    protected void print_number_range(final char sepchar, int a, int b) {
188      /* Note: we can have B < A in the case of a range of no lines.
189      In this case, we should print the line number before the range,
190      which is B.  */
191      if (++b > ++a) {
192        this.outfile.print("" + a + sepchar + b);
193      } else {
194        this.outfile.print(b);
195      }
196    }
197
198    /**
199     * Divide SCRIPT into pieces by calling HUNKFUN and print each piece with
200     * PRINTFUN. Both functions take one arg, an edit script.
201     * 
202     * PRINTFUN takes a subscript which belongs together (with a null link at
203     * the end) and prints it.
204     */
205    public void print_script(final Diff.change script) {
206      setupOutput();
207      Diff.change next = script;
208
209      while (next != null) {
210        Diff.change t, end;
211
212        /* Find a set of changes that belong together.  */
213        t = next;
214        end = hunkfun(next);
215
216        /* Disconnect them from the rest of the changes,
217           making them a hunk, and remember the rest for next iteration.  */
218        next = end.link;
219        end.link = null;
220        //if (DEBUG)
221        //  debug_script(t);
222
223        /* Print this hunk.  */
224        print_hunk(t);
225
226        /* Reconnect the script so it will all be freed properly.  */
227        end.link = next;
228      }
229      this.outfile.flush();
230    }
231
232    public void setOutput(final Writer wtr) {
233      this.outfile = new PrintWriter(wtr);
234    }
235
236    protected void setupOutput() {
237      if (this.outfile == null) {
238        this.outfile = new PrintWriter(new OutputStreamWriter(System.out));
239      }
240    }
241  }

243  /**
244   * Prints an edit script in context diff format. This and its 'unified'
245   * variation is used for source code patches.
246   */
247  public static class ContextPrint extends Base {
248
249    protected int context = 3;
250
251    public ContextPrint(final Object[] a, final Object[] b) {
252      super(a, b);
253    }
254
255    /** If function_regexp defined, search for start of function. */
256    private String find_function(final Object[] lines, final int start) {
257      return null;
258    }
259
260    protected void print_context_label(final String mark, final File inf,
261        final String label) {
262      setupOutput();
263      if (label != null) {
264        this.outfile.println(mark + ' ' + label);
265      } else if (inf.lastModified() > 0) {
266        // FIXME: use DateFormat to get precise format needed.
267        this.outfile.println(mark + ' ' + inf.getPath() + '\t'
268            + new Date(inf.lastModified()));
269      } else {
270        /* Don't pretend that standard input is ancient.  */
271        this.outfile.println(mark + ' ' + inf.getPath());
272      }
273    }
274
275    protected void print_function(final Object[] file, final int start) {
276      final String function = find_function(this.file0, this.first0);
277      if (function != null) {
278        this.outfile.print(" ");
279        this.outfile.print((function.length() < 40) ? function : function
280            .substring(0, 40));
281      }
282    }
283
284    @Override
285    public void print_header(final String filea, final String fileb) {
286      print_context_label("***", new File(filea), filea);
287      print_context_label("---", new File(fileb), fileb);
288    }
289
290    @Override
291    protected void print_hunk(final Diff.change hunk) {
292
293      /* Determine range of line numbers involved in each file.  */
294
295      analyze_hunk(hunk);
296
297      if ((this.deletes == 0) && (this.inserts == 0)) {
298        return;
299      }
300
301      /* Include a context's width before and after.  */
302
303      this.first0 = Math.max(this.first0 - this.context, 0);
304      this.first1 = Math.max(this.first1 - this.context, 0);
305      this.last0 = Math.min(this.last0 + this.context, this.file0.length - 1);
306      this.last1 = Math.min(this.last1 + this.context, this.file1.length - 1);
307
308      this.outfile.print("***************");
309
310      /* If we looked for and found a function this is part of,
311      include its name in the header of the diff section.  */
312      print_function(this.file0, this.first0);
313
314      this.outfile.println();
315      this.outfile.print("*** ");
316      print_number_range(',', this.first0, this.last0);
317      this.outfile.println(" ****");
318
319      if (this.deletes != 0) {
320        Diff.change next = hunk;
321
322        for (int i = this.first0; i <= this.last0; i++) {
323          /* Skip past changes that apply (in file 0)
324             only to lines before line I.  */
325
326          while ((next != null) && (next.line0 + next.deleted <= i)) {
327            next = next.link;
328          }
329
330          /* Compute the marking for line I.  */
331
332          String prefix = " ";
333          if ((next != null) && (next.line0 <= i)) {
334            /* The change NEXT covers this line.
335               If lines were inserted here in file 1, this is "changed".
336               Otherwise it is "deleted".  */
337            prefix = (next.inserted > 0) ? "!" : "-";
338          }
339
340          print_1_line(prefix, this.file0[i]);
341        }
342      }
343
344      this.outfile.print("--- ");
345      print_number_range(',', this.first1, this.last1);
346      this.outfile.println(" ----");
347
348      if (this.inserts != 0) {
349        Diff.change next = hunk;
350
351        for (int i = this.first1; i <= this.last1; i++) {
352          /* Skip past changes that apply (in file 1)
353             only to lines before line I.  */
354
355          while ((next != null) && (next.line1 + next.inserted <= i)) {
356            next = next.link;
357          }
358
359          /* Compute the marking for line I.  */
360
361          String prefix = " ";
362          if ((next != null) && (next.line1 <= i)) {
363            /* The change NEXT covers this line.
364               If lines were deleted here in file 0, this is "changed".
365               Otherwise it is "inserted".  */
366            prefix = (next.deleted > 0) ? "!" : "+";
367          }
368
369          print_1_line(prefix, this.file1[i]);
370        }
371      }
372    }
373  }

375  /**
376   * Prints an edit script in a format suitable for input to <code>ed</code>.
377   * The edit script must be generated with the reverse option to be useful as
378   * actual <code>ed</code> input.
379   */
380  public static class EdPrint extends Base {
381
382    public EdPrint(final Object[] a, final Object[] b) {
383      super(a, b);
384    }
385
386    /** Print a hunk of an ed diff */
387    @Override
388    protected void print_hunk(final Diff.change hunk) {
389
390      /* Determine range of line numbers involved in each file.  */
391      analyze_hunk(hunk);
392      if ((this.deletes == 0) && (this.inserts == 0)) {
393        return;
394      }
395
396      /* Print out the line number header for this hunk */
397      print_number_range(',', this.first0, this.last0);
398      this.outfile.println(Base.change_letter(this.inserts, this.deletes));
399
400      /* Print new/changed lines from second file, if needed */
401      if (this.inserts != 0) {
402        boolean inserting = true;
403        for (int i = this.first1; i <= this.last1; i++) {
404          /* Resume the insert, if we stopped.  */
405          if (!inserting) {
406            this.outfile.println(i - this.first1 + this.first0 + "a");
407          }
408          inserting = true;
409
410          /* If the file's line is just a dot, it would confuse `ed'.
411          So output it with a double dot, and set the flag LEADING_DOT
412          so that we will output another ed-command later
413          to change the double dot into a single dot.  */
414
415          if (".".equals(this.file1[i])) {
416            this.outfile.println("..");
417            this.outfile.println(".");
418            /* Now change that double dot to the desired single dot.  */
419            this.outfile.println(i - this.first1 + this.first0 + 1
420                + "s/^\\.\\././");
421            inserting = false;
422          } else {
423            /* Line is not `.', so output it unmodified.  */
424            print_1_line("", this.file1[i]);
425          }
426        }
427
428        /* End insert mode, if we are still in it.  */
429        if (inserting) {
430          this.outfile.println(".");
431        }
432      }
433    }
434  }

436  /**
437   * Print a change list in the standard diff format.
438   */
439  public static class NormalPrint extends Base {
440
441    public NormalPrint(final Object[] a, final Object[] b) {
442      super(a, b);
443    }
444
445    /**
446     * Print a hunk of a normal diff. This is a contiguous portion of a complete
447     * edit script, describing changes in consecutive lines.
448     */
449
450    @Override
451    protected void print_hunk(final Diff.change hunk) {
452
453      /* Determine range of line numbers involved in each file.  */
454      analyze_hunk(hunk);
455      if ((this.deletes == 0) && (this.inserts == 0)) {
456        return;
457      }
458
459      /* Print out the line number header for this hunk */
460      print_number_range(',', this.first0, this.last0);
461      this.outfile.print(Base.change_letter(this.inserts, this.deletes));
462      print_number_range(',', this.first1, this.last1);
463      this.outfile.println();
464
465      /* Print the lines that the first file has.  */
466      if (this.deletes != 0) {
467        for (int i = this.first0; i <= this.last0; i++) {
468          print_1_line("< ", this.file0[i]);
469        }
470      }
471
472      if ((this.inserts != 0) && (this.deletes != 0)) {
473        this.outfile.println("---");
474      }
475
476      /* Print the lines that the second file has.  */
477      if (this.inserts != 0) {
478        for (int i = this.first1; i <= this.last1; i++) {
479          print_1_line("> ", this.file1[i]);
480        }
481      }
482    }
483  }

485  /**
486   * Prints an edit script in context diff format. This and its 'unified'
487   * variation is used for source code patches.
488   */
489  public static class UnifiedPrint extends ContextPrint {
490
491    public UnifiedPrint(final Object[] a, final Object[] b) {
492      super(a, b);
493    }
494
495    @Override
496    public void print_header(final String filea, final String fileb) {
497      print_context_label("---", new File(filea), filea);
498      print_context_label("+++", new File(fileb), fileb);
499    }
500
501    @Override
502    protected void print_hunk(final Diff.change hunk) {
503      /* Determine range of line numbers involved in each file.  */
504      analyze_hunk(hunk);
505
506      if ((this.deletes == 0) && (this.inserts == 0)) {
507        return;
508      }
509
510      /* Include a context's width before and after.  */
511
512      this.first0 = Math.max(this.first0 - this.context, 0);
513      this.first1 = Math.max(this.first1 - this.context, 0);
514      this.last0 = Math.min(this.last0 + this.context, this.file0.length - 1);
515      this.last1 = Math.min(this.last1 + this.context, this.file1.length - 1);
516
517      this.outfile.print("@@ -");
518      print_number_range(this.first0, this.last0);
519      this.outfile.print(" +");
520      print_number_range(this.first1, this.last1);
521      this.outfile.print(" @@");
522
523      /* If we looked for and found a function this is part of,
524      include its name in the header of the diff section.  */
525      print_function(this.file0, this.first0);
526
527      this.outfile.println();
528
529      Diff.change next = hunk;
530      int i = this.first0;
531      int j = this.first1;
532
533      while ((i <= this.last0) || (j <= this.last1)) {
534
535        /* If the line isn't a difference, output the context from file 0. */
536
537        if ((next == null) || (i < next.line0)) {
538          if (i < this.file0.length) {
539            this.outfile.print(' ');
540            print_1_line("", this.file0[i++]);
541          }
542          j++;
543        } else {
544          /* For each difference, first output the deleted part. */
545
546          int k = next.deleted;
547          while (k-- > 0) {
548            this.outfile.print('-');
549            print_1_line("", this.file0[i++]);
550          }
551
552          /* Then output the inserted part. */
553
554          k = next.inserted;
555          while (k-- > 0) {
556            this.outfile.print('+');
557            print_1_line("", this.file1[j++]);
558          }
559
560          /* We're done with this hunk, so on to the next! */
561
562          next = next.link;
563        }
564      }
565    }
566
567    private void print_number_range(final int a, final int b) {
568      //translate_range (file, a, b, &trans_a, &trans_b);
569
570      /* Note: we can have B < A in the case of a range of no lines.
571      In this case, we should print the line number before the range,
572      which is B.  */
573      if (b < a) {
574        this.outfile.print(b + ",0");
575      } else {
576        super.print_number_range(',', a, b);
577      }
578    }
579  }

581  public static void main(final String[] argv) throws IOException {
582    final String filea = argv[argv.length - 2];
583    final String fileb = argv[argv.length - 1];
584    final String[] a = DiffPrint.slurp(filea);
585    final String[] b = DiffPrint.slurp(fileb);
586    final Diff d = new Diff(a, b);
587    char style = 'n';
588    for (int i = 0; i < argv.length - 2; ++i) {
589      final String f = argv[i];
590      if (f.startsWith("-")) {
591        for (int j = 1; j < f.length(); ++j) {
592          switch (f.charAt(j)) {
593            case 'e': // Ed style
594              style = 'e';
595              break;
596            case 'c': // Context diff
597              style = 'c';
598              break;
599            case 'u':
600              style = 'u';
601              break;
602          }
603        }
604      }
605    }
606    final boolean reverse = style == 'e';
607    final Diff.change script = d.diff_2(reverse);
608    if (script == null) {
609      System.err.println("No differences");
610    } else {
611      Base p;
612      switch (style) {
613        case 'e':
614          p = new EdPrint(a, b);
615          break;
616        case 'c':
617          p = new ContextPrint(a, b);
618          break;
619        case 'u':
620          p = new UnifiedPrint(a, b);
621          break;
622        default:
623          p = new NormalPrint(a, b);
624      }
625      p.print_header(filea, fileb);
626      p.print_script(script);
627    }
628  }

630  /**
631   * Read a text file into an array of String. This provides basic diff
632   * functionality. A more advanced diff utility will use specialized objects to
633   * represent the text lines, with options to, for example, convert sequences
634   * of whitespace to a single space for comparison purposes.
635   */
636  @SuppressWarnings({ "rawtypes", "resource", "unchecked" })
637  static String[] slurp(final String file) throws IOException {
638    final BufferedReader rdr = new BufferedReader(new FileReader(file));
639    final Vector s = new Vector();
640    for (;;) {
641      final String line = rdr.readLine();
642      if (line == null) {
643        break;
644      }
645      s.addElement(line);
646    }
647    final String[] a = new String[s.size()];
648    s.copyInto(a);
649    return a;
650  }
