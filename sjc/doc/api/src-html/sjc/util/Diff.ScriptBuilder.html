040/**
041 * A class to compare vectors of objects. The result of comparison is a list of
042 * <code>change</code> objects which form an edit script. The objects compared
043 * are traditionally lines of text from two files. Comparison options such as
044 * "ignore whitespace" are implemented by modifying the <code>equals</code> and
045 * <code>hashcode</code> methods for the objects compared.
046 * <p>
047 * The basic algorithm is described in: </br>
048 * "An O(ND) Difference Algorithm and its Variations", Eugene Myers,
049 * Algorithmica Vol. 1 No. 2, 1986, p 251.
050 * <p>
051 * This class outputs different results from GNU diff 1.15 on some inputs. Our
052 * results are actually better (smaller change list, smaller total size of
053 * changes), but it would be nice to know why. Perhaps there is a memory
054 * overwrite bug in GNU diff 1.15.
055 *

057 *         Business Management Systems, Inc.

075public class Diff {

077  /**
078   * The result of comparison is an "edit script": a chain of change objects.
079   * Each change represents one place where some lines are deleted and some are
080   * inserted.
081   * 
082   * LINE0 and LINE1 are the first affected lines in the two files (origin 0).
083   * DELETED is the number of lines deleted here from file 0. INSERTED is the
084   * number of lines inserted here in file 1.
085   * 
086   * If DELETED is 0 then LINE0 is the number of the line before which the
087   * insertion was done; vice versa for INSERTED and LINE1.
088   */
089
090  public static class change {
091    /** Previous or next edit command. */
092    public change link;
093    /** # lines of file 1 changed here. */
094    public final int inserted;
095    /** # lines of file 0 changed here. */
096    public final int deleted;
097    /** Line number of 1st deleted line. */
098    public final int line0;
099    /** Line number of 1st inserted line. */
100    public final int line1;
101
102    /**
103     * Cons an additional entry onto the front of an edit script OLD. LINE0 and
104     * LINE1 are the first affected lines in the two files (origin 0). DELETED
105     * is the number of lines deleted here from file 0. INSERTED is the number
106     * of lines inserted here in file 1.
107     * 
108     * If DELETED is 0 then LINE0 is the number of the line before which the
109     * insertion was done; vice versa for INSERTED and LINE1.
110     */
111    public change(final int line0, final int line1, final int deleted,
112        final int inserted, final change old) {
113      this.line0 = line0;
114      this.line1 = line1;
115      this.inserted = inserted;
116      this.deleted = deleted;
117      this.link = old;
118      //System.err.println(line0+","+line1+","+inserted+","+deleted);
119    }
120  }

122  /**
123   * Data on one input file being compared.
124   */
125
126  class file_data {
127
128    /** Number of elements (lines) in this file. */
129    final int buffered_lines;
130
131    /**
132     * Vector, indexed by line number, containing an equivalence code for each
133     * line. It is this vector that is actually compared with that of another
134     * file to generate differences.
135     */
136    private final int[] equivs;
137
138    /**
139     * Vector, like the previous one except that the elements for discarded
140     * lines have been squeezed out.
141     */
142    final int[] undiscarded;
143
144    /**
145     * Vector mapping virtual line numbers (not counting discarded lines) to
146     * real ones (counting those lines). Both are origin-0.
147     */
148    final int[] realindexes;
149
150    /** Total number of nondiscarded lines. */
151    int nondiscarded_lines;
152
153    /**
154     * Array, indexed by real origin-1 line number, containing true for a line
155     * that is an insertion or a deletion. The results of comparison are stored
156     * here.
157     */
158    boolean[] changed_flag;
159
160    @SuppressWarnings({ "rawtypes", "unchecked" })
161    file_data(final Object[] data, final Hashtable h) {
162      this.buffered_lines = data.length;
163
164      this.equivs = new int[this.buffered_lines];
165      this.undiscarded = new int[this.buffered_lines];
166      this.realindexes = new int[this.buffered_lines];
167
168      for (int i = 0; i < data.length; ++i) {
169        final Integer ir = (Integer) h.get(data[i]);
170        if (ir == null) {
171          h.put(data[i], this.equivs[i] = Diff.this.equiv_max++);
172        } else {
173          this.equivs[i] = ir;
174        }
175      }
176    }
177
178    /** Allocate changed array for the results of comparison. */
179    void clear() {
180      /* Allocate a flag for each line of each file, saying whether that line
181      is an insertion or deletion.
182      Allocate an extra element, always zero, at each end of each vector.
183       */
184      this.changed_flag = new boolean[this.buffered_lines + 2];
185    }
186
187    /**
188     * Actually discard the lines.
189     * 
190     * @param discards
191     *          flags lines to be discarded
192     */
193    private void discard(final byte[] discards) {
194      int j = 0;
195      for (int i = 0; i < this.buffered_lines; ++i) {
196        if (Diff.this.no_discards || (discards[i] == 0)) {
197          this.undiscarded[j] = this.equivs[i];
198          this.realindexes[j++] = i;
199        } else {
200          this.changed_flag[1 + i] = true;
201        }
202      }
203      this.nondiscarded_lines = j;
204    }
205
206    /**
207     * Discard lines that have no matches in another file.
208     * 
209     * A line which is discarded will not be considered by the actual comparison
210     * algorithm; it will be as if that line were not in the file. The file's
211     * `realindexes' table maps virtual line numbers (which don't count the
212     * discarded lines) into real line numbers; this is how the actual
213     * comparison algorithm produces results that are comprehensible when the
214     * discarded lines are counted.
215     * <p>
216     * When we discard a line, we also mark it as a deletion or insertion so
217     * that it will be printed in the output.
218     * 
219     * @param f
220     *          the other file
221     */
222    void discard_confusing_lines(final file_data f) {
223      clear();
224      /* Set up table of which lines are going to be discarded. */
225      final byte[] discarded = discardable(f.equivCount());
226
227      /* Don't really discard the provisional lines except when they occur
228         in a run of discardables, with nonprovisionals at the beginning
229         and end.  */
230      filterDiscards(discarded);
231
232      /* Actually discard the lines. */
233      discard(discarded);
234    }
235
236    /**
237     * Mark to be discarded each line that matches no line of another file. If a
238     * line matches many lines, mark it as provisionally discardable.
239     * 
240     * @see equivCount()
241     * @param counts
242     *          The count of each equivalence number for the other file.
243     * @return 0=nondiscardable, 1=discardable or 2=provisionally discardable
244     *         for each line
245     */
246
247    private byte[] discardable(final int[] counts) {
248      final int end = this.buffered_lines;
249      final byte[] discards = new byte[end];
250      final int[] equivs = this.equivs;
251      int many = 5;
252      int tem = end / 64;
253
254      /* Multiply MANY by approximate square root of number of lines.
255      That is the threshold for provisionally discardable lines.  */
256      while ((tem = tem >> 2) > 0) {
257        many *= 2;
258      }
259
260      for (int i = 0; i < end; i++) {
261        int nmatch;
262        if (equivs[i] == 0) {
263          continue;
264        }
265        nmatch = counts[equivs[i]];
266        if (nmatch == 0) {
267          discards[i] = 1;
268        } else if (nmatch > many) {
269          discards[i] = 2;
270        }
271      }
272      return discards;
273    }
274
275    /**
276     * Return equiv_count[I] as the number of lines in this file that fall in
277     * equivalence class I.
278     * 
279     * @return the array of equivalence class counts.
280     */
281    int[] equivCount() {
282      final int[] equiv_count = new int[Diff.this.equiv_max];
283      for (int i = 0; i < this.buffered_lines; ++i) {
284        ++equiv_count[this.equivs[i]];
285      }
286      return equiv_count;
287    }
288
289    /**
290     * Don't really discard the provisional lines except when they occur in a
291     * run of discardables, with nonprovisionals at the beginning and end.
292     */
293
294    private void filterDiscards(final byte[] discards) {
295      final int end = this.buffered_lines;
296
297      for (int i = 0; i < end; i++) {
298        /* Cancel provisional discards not in middle of run of discards.  */
299        if (discards[i] == 2) {
300          discards[i] = 0;
301        } else if (discards[i] != 0) {
302          /* We have found a nonprovisional discard.  */
303          int j;
304          int length;
305          int provisional = 0;
306
307          /* Find end of this run of discardable lines.
308             Count how many are provisionally discardable.  */
309          for (j = i; j < end; j++) {
310            if (discards[j] == 0) {
311              break;
312            }
313            if (discards[j] == 2) {
314              ++provisional;
315            }
316          }
317
318          /* Cancel provisional discards at end, and shrink the run.  */
319          while ((j > i) && (discards[j - 1] == 2)) {
320            discards[--j] = 0;
321            --provisional;
322          }
323
324          /* Now we have the length of a run of discardable lines
325             whose first and last are not provisional.  */
326          length = j - i;
327
328          /* If 1/4 of the lines in the run are provisional,
329             cancel discarding of all provisional lines in the run.  */
330          if (provisional * 4 > length) {
331            while (j > i) {
332              if (discards[--j] == 2) {
333                discards[j] = 0;
334              }
335            }
336          } else {
337            int consec;
338            int minimum = 1;
339            int tem = length / 4;
340
341            /* MINIMUM is approximate square root of LENGTH/4.
342               A subrun of two or more provisionals can stand
343               when LENGTH is at least 16.
344               A subrun of 4 or more can stand when LENGTH >= 64.  */
345            while ((tem = tem >> 2) > 0) {
346              minimum *= 2;
347            }
348            minimum++;
349
350            /* Cancel any subrun of MINIMUM or more provisionals
351               within the larger run.  */
352            for (j = 0, consec = 0; j < length; j++) {
353              if (discards[i + j] != 2) {
354                consec = 0;
355              } else if (minimum == ++consec) {
356                /* Back up to start of subrun, to cancel it all.  */
357                j -= consec;
358              } else if (minimum < consec) {
359                discards[i + j] = 0;
360              }
361            }
362
363            /* Scan from beginning of run
364               until we find 3 or more nonprovisionals in a row
365               or until the first nonprovisional at least 8 lines in.
366               Until that point, cancel any provisionals.  */
367            for (j = 0, consec = 0; j < length; j++) {
368              if ((j >= 8) && (discards[i + j] == 1)) {
369                break;
370              }
371              if (discards[i + j] == 2) {
372                consec = 0;
373                discards[i + j] = 0;
374              } else if (discards[i + j] == 0) {
375                consec = 0;
376              } else {
377                consec++;
378              }
379              if (consec == 3) {
380                break;
381              }
382            }
383
384            /* I advances to the last line of the run.  */
385            i += length - 1;
386
387            /* Same thing, from end.  */
388            for (j = 0, consec = 0; j < length; j++) {
389              if ((j >= 8) && (discards[i - j] == 1)) {
390                break;
391              }
392              if (discards[i - j] == 2) {
393                consec = 0;
394                discards[i - j] = 0;
395              } else if (discards[i - j] == 0) {
396                consec = 0;
397              } else {
398                consec++;
399              }
400              if (consec == 3) {
401                break;
402              }
403            }
404          }
405        }
406      }
407    }

409    /**
410     * Adjust inserts/deletes of blank lines to join changes as much as
411     * possible.
412     * 
413     * We do something when a run of changed lines include a blank line at one
414     * end and have an excluded blank line at the other. We are free to choose
415     * which blank line is included. `compareseq' always chooses the one at the
416     * beginning, but usually it is cleaner to consider the following blank line
417     * to be the "change". The only exception is if the preceding blank line
418     * would join this change to other changes.
419     * 
420     * @param f
421     *          the file being compared against
422     */
423
424    void shift_boundaries(final file_data f) {
425      final boolean[] changed = this.changed_flag;
426      final boolean[] other_changed = f.changed_flag;
427      int i = 0;
428      int j = 0;
429      final int i_end = this.buffered_lines;
430      int preceding = -1;
431      int other_preceding = -1;
432
433      for (;;) {
434        int start, end, other_start;
435
436        /* Scan forwards to find beginning of another run of changes.
437           Also keep track of the corresponding point in the other file.  */
438
439        while ((i < i_end) && !changed[1 + i]) {
440          while (other_changed[1 + j++]) {
441            /* Non-corresponding lines in the other file
442               will count as the preceding batch of changes.  */
443            other_preceding = j;
444          }
445          i++;
446        }
447
448        if (i == i_end) {
449          break;
450        }
451
452        start = i;
453        other_start = j;
454
455        for (;;) {
456          /* Now find the end of this run of changes.  */
457
458          while ((i < i_end) && changed[1 + i]) {
459            i++;
460          }
461          end = i;
462
463          /* If the first changed line matches the following unchanged one,
464          and this run does not follow right after a previous run,
465          and there are no lines deleted from the other file here,
466          then classify the first changed line as unchanged
467          and the following line as changed in its place.  */
468
469          /* You might ask, how could this run follow right after another?
470          Only because the previous run was shifted here.  */
471
472          if ((end != i_end)
473              && (this.equivs[start] == this.equivs[end])
474              && !other_changed[1 + j]
475              && (end != i_end)
476              && !(((preceding >= 0) && (start == preceding)) || ((other_preceding >= 0) && (other_start == other_preceding)))) {
477            changed[1 + end++] = true;
478            changed[1 + start++] = false;
479            ++i;
480            /* Since one line-that-matches is now before this run
481               instead of after, we must advance in the other file
482               to keep in synch.  */
483            ++j;
484          } else {
485            break;
486          }
487        }
488
489        preceding = i;
490        other_preceding = j;
491      }
492    }
493
494  }

496  static class ForwardScript
497      implements ScriptBuilder {
498    /**
499     * Scan the tables of which lines are inserted and deleted, producing an
500     * edit script in forward order.
501     */
502    public change build_script(final boolean[] changed0, final int len0,
503        final boolean[] changed1, final int len1) {
504      change script = null;
505      int i0 = len0, i1 = len1;
506
507      while ((i0 >= 0) || (i1 >= 0)) {
508        if (changed0[i0] || changed1[i1]) {
509          final int line0 = i0, line1 = i1;
510
511          /* Find # lines changed here in each file.  */
512          while (changed0[i0]) {
513            --i0;
514          }
515          while (changed1[i1]) {
516            --i1;
517          }
518
519          /* Record this change.  */
520          script = new change(i0, i1, line0 - i0, line1 - i1, script);
521        }
522
523        /* We have reached lines in the two files that match each other.  */
524        i0--;
525        i1--;
526      }
527
528      return script;
529    }
530  }

532  /**
533   * Scan the tables of which lines are inserted and deleted, producing an edit
534   * script in reverse order.

537  static class ReverseScript
538      implements ScriptBuilder {
539    public change build_script(final boolean[] changed0, final int len0,
540        final boolean[] changed1, final int len1) {
541      change script = null;
542      int i0 = 0, i1 = 0;
543      while ((i0 < len0) || (i1 < len1)) {
544        if (changed0[1 + i0] || changed1[1 + i1]) {
545          final int line0 = i0, line1 = i1;
546
547          /* Find # lines changed here in each file.  */
548          while (changed0[1 + i0]) {
549            ++i0;
550          }
551          while (changed1[1 + i1]) {
552            ++i1;
553          }
554
555          /* Record this change.  */
556          script = new change(line0, line1, i0 - line0, i1 - line1, script);
557        }
558
559        /* We have reached lines in the two files that match each other.  */
560        i0++;
561        i1++;
562      }
563
564      return script;
565    }
566  }

626  /**
627   * Prepare to find differences between two arrays. Each element of the arrays
628   * is translated to an "equivalence number" based on the result of
629   * <code>equals</code>. The original Object arrays are no longer needed for
630   * computing the differences. They will be needed again later to print the
631   * results of the comparison as an edit script, if desired.
632   */
633  @SuppressWarnings("rawtypes")
634  public Diff(final Object[] a, final Object[] b) {
635    final Hashtable h = new Hashtable(a.length + b.length);
636    this.filevec[0] = new file_data(a, h);
637    this.filevec[1] = new file_data(b, h);
638  }
639
640  /**
641   * Compare in detail contiguous subsequences of the two files which are known,
642   * as a whole, to match each other.
643   * 
644   * The results are recorded in the vectors filevec[N].changed_flag, by storing
645   * a 1 in the element for each line that is an insertion or deletion.
646   * 
647   * The subsequence of file 0 is [XOFF, XLIM) and likewise for file 1.
648   * 
649   * Note that XLIM, YLIM are exclusive bounds. All line numbers are origin-0
650   * and discarded lines are not counted.
651   */
652
653  private void compareseq(int xoff, int xlim, int yoff, int ylim) {
654    /* Slide down the bottom initial diagonal. */
655    while ((xoff < xlim) && (yoff < ylim)
656        && (this.xvec[xoff] == this.yvec[yoff])) {
657      ++xoff;
658      ++yoff;
659    }
660    /* Slide up the top initial diagonal. */
661    while ((xlim > xoff) && (ylim > yoff)
662        && (this.xvec[xlim - 1] == this.yvec[ylim - 1])) {
663      --xlim;
664      --ylim;
665    }
666
667    /* Handle simple cases. */
668    if (xoff == xlim) {
669      while (yoff < ylim) {
670        this.filevec[1].changed_flag[1 + this.filevec[1].realindexes[yoff++]] = true;
671      }
672    } else if (yoff == ylim) {
673      while (xoff < xlim) {
674        this.filevec[0].changed_flag[1 + this.filevec[0].realindexes[xoff++]] = true;
675      }
676    } else {
677      /* Find a point of correspondence in the middle of the files.  */
678
679      final int d = diag(xoff, xlim, yoff, ylim);
680      final int c = this.cost;
681      final int b = this.bdiag[this.bdiagoff + d];
682
683      if (c == 1) {
684        /* This should be impossible, because it implies that
685           one of the two subsequences is empty,
686           and that case was handled above without calling `diag'.
687           Let's verify that this is true.  */
688        throw new IllegalArgumentException("Empty subsequence");
689      } else {
690        /* Use that point to split this problem into two subproblems.  */
691        compareseq(xoff, b, yoff, b - d);
692        /* This used to use f instead of b,
693           but that is incorrect!
694           It is not necessarily the case that diagonal d
695           has a snake from b to f.  */
696        compareseq(b, xlim, b - d, ylim);
697      }
698    }
699  }

701  /**
702   * Find the midpoint of the shortest edit script for a specified portion of
703   * the two files.
704   * 
705   * We scan from the beginnings of the files, and simultaneously from the ends,
706   * doing a breadth-first search through the space of edit-sequence. When the
707   * two searches meet, we have found the midpoint of the shortest edit
708   * sequence.
709   * 
710   * The value returned is the number of the diagonal on which the midpoint
711   * lies. The diagonal number equals the number of inserted lines minus the
712   * number of deleted lines (counting only lines before the midpoint). The edit
713   * cost is stored into COST; this is the total number of lines inserted or
714   * deleted (counting only lines before the midpoint).
715   * 
716   * This function assumes that the first lines of the specified portions of the
717   * two files do not match, and likewise that the last lines do not match. The
718   * caller must trim matching lines from the beginning and end of the portions
719   * it is going to specify.
720   * 
721   * Note that if we return the "wrong" diagonal value, or if the value of bdiag
722   * at that diagonal is "wrong", the worst this can do is cause suboptimal diff
723   * output. It cannot cause incorrect diff output.
724   */
725
726  private int diag(final int xoff, final int xlim, final int yoff,
727      final int ylim) {
728    final int[] fd = this.fdiag; // Give the compiler a chance.
729    final int[] bd = this.bdiag; // Additional help for the compiler.
730    final int[] xv = this.xvec; // Still more help for the compiler.
731    final int[] yv = this.yvec; // And more and more . . .
732    final int dmin = xoff - ylim; // Minimum valid diagonal.
733    final int dmax = xlim - yoff; // Maximum valid diagonal.
734    final int fmid = xoff - yoff; // Center diagonal of top-down search.
735    final int bmid = xlim - ylim; // Center diagonal of bottom-up search.
736    int fmin = fmid, fmax = fmid; // Limits of top-down search.
737    int bmin = bmid, bmax = bmid; // Limits of bottom-up search.
738    /* True if southeast corner is on an odd
739                     diagonal with respect to the northwest. */
740    final boolean odd = (fmid - bmid & 1) != 0;
741
742    fd[this.fdiagoff + fmid] = xoff;
743    bd[this.bdiagoff + bmid] = xlim;
744
745    for (int c = 1;; ++c) {
746      int d; /* Active diagonal. */
747      boolean big_snake = false;
748
749      /* Extend the top-down search by an edit step in each diagonal. */
750      if (fmin > dmin) {
751        fd[this.fdiagoff + --fmin - 1] = -1;
752      } else {
753        ++fmin;
754      }
755      if (fmax < dmax) {
756        fd[this.fdiagoff + ++fmax + 1] = -1;
757      } else {
758        --fmax;
759      }
760      for (d = fmax; d >= fmin; d -= 2) {
761        int x, y, oldx;
762        final int tlo = fd[this.fdiagoff + d - 1], thi = fd[this.fdiagoff + d
763            + 1];
764
765        if (tlo >= thi) {
766          x = tlo + 1;
767        } else {
768          x = thi;
769        }
770        oldx = x;
771        y = x - d;
772        while ((x < xlim) && (y < ylim) && (xv[x] == yv[y])) {
773          ++x;
774          ++y;
775        }
776        if (x - oldx > 20) {
777          big_snake = true;
778        }
779        fd[this.fdiagoff + d] = x;
780        if (odd && (bmin <= d) && (d <= bmax)
781            && (bd[this.bdiagoff + d] <= fd[this.fdiagoff + d])) {
782          this.cost = 2 * c - 1;
783          return d;
784        }
785      }
786
787      /* Similar extend the bottom-up search. */
788      if (bmin > dmin) {
789        bd[this.bdiagoff + --bmin - 1] = Integer.MAX_VALUE;
790      } else {
791        ++bmin;
792      }
793      if (bmax < dmax) {
794        bd[this.bdiagoff + ++bmax + 1] = Integer.MAX_VALUE;
795      } else {
796        --bmax;
797      }
798      for (d = bmax; d >= bmin; d -= 2) {
799        int x, y, oldx;
800        final int tlo = bd[this.bdiagoff + d - 1], thi = bd[this.bdiagoff + d
801            + 1];
802
803        if (tlo < thi) {
804          x = tlo;
805        } else {
806          x = thi - 1;
807        }
808        oldx = x;
809        y = x - d;
810        while ((x > xoff) && (y > yoff) && (xv[x - 1] == yv[y - 1])) {
811          --x;
812          --y;
813        }
814        if (oldx - x > 20) {
815          big_snake = true;
816        }
817        bd[this.bdiagoff + d] = x;
818        if (!odd && (fmin <= d) && (d <= fmax)
819            && (bd[this.bdiagoff + d] <= fd[this.fdiagoff + d])) {
820          this.cost = 2 * c;
821          return d;
822        }
823      }
824
825      /* Heuristic: check occasionally for a diagonal that has made
826         lots of progress compared with the edit distance.
827         If we have any such, find the one that has made the most
828         progress and return it as if it had succeeded.
829
830         With this heuristic, for files with a constant small density
831         of changes, the algorithm is linear in the file size.  */
832
833      if ((c > 200) && big_snake && this.heuristic) {
834        int best = 0;
835        int bestpos = -1;
836
837        for (d = fmax; d >= fmin; d -= 2) {
838          final int dd = d - fmid;
839          if ((fd[this.fdiagoff + d] - xoff) * 2 - dd > 12 * (c + (dd > 0 ? dd
840              : -dd))) {
841            if ((fd[this.fdiagoff + d] * 2 - dd > best)
842                && (fd[this.fdiagoff + d] - xoff > 20)
843                && (fd[this.fdiagoff + d] - d - yoff > 20)) {
844              int k;
845              final int x = fd[this.fdiagoff + d];
846
847              /* We have a good enough best diagonal;
848                 now insist that it end with a significant snake.  */
849              for (k = 1; k <= 20; k++) {
850                if (this.xvec[x - k] != this.yvec[x - d - k]) {
851                  break;
852                }
853              }
854
855              if (k == 21) {
856                best = fd[this.fdiagoff + d] * 2 - dd;
857                bestpos = d;
858              }
859            }
860          }
861        }
862        if (best > 0) {
863          this.cost = 2 * c - 1;
864          return bestpos;
865        }
866
867        best = 0;
868        for (d = bmax; d >= bmin; d -= 2) {
869          final int dd = d - bmid;
870          if ((xlim - bd[this.bdiagoff + d]) * 2 + dd > 12 * (c + (dd > 0 ? dd
871              : -dd))) {
872            if (((xlim - bd[this.bdiagoff + d]) * 2 + dd > best)
873                && (xlim - bd[this.bdiagoff + d] > 20)
874                && (ylim - (bd[this.bdiagoff + d] - d) > 20)) {
875              /* We have a good enough best diagonal;
876                 now insist that it end with a significant snake.  */
877              int k;
878              final int x = bd[this.bdiagoff + d];
879
880              for (k = 0; k < 20; k++) {
881                if (this.xvec[x + k] != this.yvec[x - d + k]) {
882                  break;
883                }
884              }
885              if (k == 20) {
886                best = (xlim - bd[this.bdiagoff + d]) * 2 + dd;
887                bestpos = d;
888              }
889            }
890          }
891        }
892        if (best > 0) {
893          this.cost = 2 * c - 1;
894          return bestpos;
895        }
896      }
897    }
898  }

900  /**
901   * Get the results of comparison as an edit script. The script is described by
902   * a list of changes. The standard ScriptBuilder implementations provide for
903   * forward and reverse edit scripts. Alternate implementations could, for
904   * instance, list common elements instead of differences.
905   * 
906   * @param bld
907   *          an object to build the script from change flags
908   * @return the head of a list of changes
909   */
910  public change diff(final ScriptBuilder bld) {
911
912    /* Some lines are obviously insertions or deletions
913       because they don't match anything.  Detect them now,
914       and avoid even thinking about them in the main comparison algorithm.  */
915
916    discard_confusing_lines();
917
918    /* Now do the main comparison algorithm, considering just the
919       undiscarded lines.  */
920
921    this.xvec = this.filevec[0].undiscarded;
922    this.yvec = this.filevec[1].undiscarded;
923
924    final int diags = this.filevec[0].nondiscarded_lines
925        + this.filevec[1].nondiscarded_lines + 3;
926    this.fdiag = new int[diags];
927    this.fdiagoff = this.filevec[1].nondiscarded_lines + 1;
928    this.bdiag = new int[diags];
929    this.bdiagoff = this.filevec[1].nondiscarded_lines + 1;
930
931    compareseq(0, this.filevec[0].nondiscarded_lines, 0,
932        this.filevec[1].nondiscarded_lines);
933    this.fdiag = null;
934    this.bdiag = null;
935
936    /* Modify the results slightly to make them prettier
937       in cases where that can validly be done.  */
938
939    shift_boundaries();
940
941    /* Get the results of comparison in the form of a chain
942       of `struct change's -- an edit script.  */
943    return bld.build_script(this.filevec[0].changed_flag,
944        this.filevec[0].buffered_lines, this.filevec[1].changed_flag,
945        this.filevec[1].buffered_lines);
946
947  }
948
949  /* Report the differences of two files.  DEPTH is the current directory
950     depth. */
951  public final change diff_2(final boolean reverse) {
952    return diff(reverse ? Diff.reverseScript : Diff.forwardScript);
953  }

955  /**
956   * Discard lines from one file that have no matches in the other file.
957   */
958
959  private void discard_confusing_lines() {
960    this.filevec[0].discard_confusing_lines(this.filevec[1]);
961    this.filevec[1].discard_confusing_lines(this.filevec[0]);
962  }
963
964  /**
965   * Adjust inserts/deletes of blank lines to join changes as much as possible.
966   */
967
968  private void shift_boundaries() {
969    if (this.inhibit) {
970      return;
971    }
972    this.filevec[0].shift_boundaries(this.filevec[1]);
973    this.filevec[1].shift_boundaries(this.filevec[0]);
974  }
