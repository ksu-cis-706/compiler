001package sjc.util;
002
003import sjc.annotation.PossiblyNull;
004
005/**
006 * This interface represents a function that process an input of type I.
007 * 
008 * @author <a href="mailto:robby@ksu.edu">Robby</a>
009 */
010public interface VoidFunction<I> {
011  /**
012   * Apply this function to process a given input.
013   * @param i The input.
014   */
015  void apply(@PossiblyNull I i);
016}
